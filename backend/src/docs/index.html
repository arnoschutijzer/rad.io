# authenticating

/register

POST (json/application)

{ "username": "a_username", "email": "an_em@il.com", "password": "a_password" }




200: all good

{ "success": true, "message": "Registered!" }

409: conflict

{ "success": false, "message": "This e-mail is already registered!" }

/login

POST (json/application)

{ "email": "an_em@il.com", "password": "a_password" }




200: all good

{ "success": true, "token": "JWT {token}", "user": { "_id": "{id}", "email": "an_em@il.com", "__v": 0 } }

401: unauthorized

{ "success": false, "message": "Incorrect username/password." }

/checkToken

POST (json/application)

HEADERS: { Authorization: "JWT {token}" }




200: all good

{ "success": true }

401: unauthorized
